using UnityEngine;
using SerializeInterface;
public partial class InterfacePrefabConsumer : MonoBehaviour, ISerializationCallbackReceiver
{

    [UnityEngine.TooltipAttribute("This is a tooltip")]
[SerializeField,ValidateInterface(typeof(SerializeInterface.Samples.IFoo))]
    private UnityEngine.Object _fooPrefabSerialized;
    [SerializeField,ValidateInterface(typeof(SerializeInterface.Samples.IBar))]
    private UnityEngine.Object _barPrefabSerialized;

    void ISerializationCallbackReceiver.OnBeforeSerialize()
    {
    }

    void ISerializationCallbackReceiver.OnAfterDeserialize()
    {

    _fooPrefab = _fooPrefabSerialized as SerializeInterface.Samples.IFoo;
    _barPrefab = _barPrefabSerialized as SerializeInterface.Samples.IBar;

    }

}
